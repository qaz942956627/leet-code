class Solution {
    public int deepestLeavesSum(TreeNode root) {
        if (root == null) return 0;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);

        int sum = 0;
        while (!q.isEmpty()) {
            sum = 0;
            int sz = q.size();
            for (int i = 0; i < sz; i++) {
                TreeNode cur = q.poll();
                // ç´¯åŠ ä¸€å±‚çš„èŠ‚ç‚¹ä¹‹å’Œ
                sum += cur.val;
                if (cur.left != null)
                    q.offer(cur.left);
                if (cur.right != null)
                    q.offer(cur.right);
            }
        }
        // ç°åœ¨å°±æ˜¯æœ€åä¸€å±‚çš„èŠ‚ç‚¹å€¼å’Œ
        return sum;
    }
